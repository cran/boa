\HeaderA{boa.print.cor}{Print Correlation Matrices}{boa.print.cor}
\keyword{print}{boa.print.cor}
\begin{Description}\relax
Iteratively computes and displays the correlation matrices for the parameters 
in each of the MCMC sequences in the working session list of sequences.
\end{Description}
\begin{Usage}
\begin{verbatim}
boa.print.cor()
\end{verbatim}
\end{Usage}
